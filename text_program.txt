Fazool calculateSum(Ginti x, Ginti y) ::
    Ginti result = x +x y!
    Dilcha bhai("Sum calculated")!
    Ikhtatam result!
;

Fazool checkCondition(Ginti num) ::
    Asay ho (num > 10) ::
        Dilcha bhai("Number is greater")!
    ;
;
